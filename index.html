function doGet() { const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("rehab"); if (!sheet) { return json({ ok: false, error: "rehabシートが見つかりません" }); } const updatedAt = sheet.getRange("B1").getDisplayValue(); const summary = sheet.getRange("A3:F5").getDisplayValues(); const staff = sheet.getRange("A7:F8").getDisplayValues(); return json({ ok: true, updatedAt: updatedAt, summary: { headers: summary[0], rows: summary.slice(1) }, staff: { headers: staff[0], rows: staff.slice(1) } }); } function json(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
